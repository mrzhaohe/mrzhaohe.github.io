# vuepress 配置 github.io

# npm、node 安装

# 1、将之前用 brew 安装的 node 先卸载掉

rm -rf /usr/local/lib/node_modules
brew uninstall node

# 2、使用 brew 安装 nvm

brew install nvm

# 3、安装成功后（会有稍长时间等待）

需要配置一下 .zshrc（没有就创建一个，目录 ~/） .nvm 文件夹需要创建的话也直接创建了（也是在 ~/）

touch ~/.zshrc

# 4、创建后（本来就有这个文件），往里面添加 nvm 的配置

# 直接拷贝就行，我是从 nvm 的 github 找到的

# 添加完之后记得 source 一下

export NVM_DIR="/Users/zhaohe/.nvm"
  [ -s "/usr/local/opt/nvm/nvm.sh" ] && . "/usr/local/opt/nvm/nvm.sh" 
  # This loads nvm
  [ -s "/usr/local/opt/nvm/etc/bash_completion.d/nvm" ] && . "/usr/local/opt/nvm/etc/bash_completion.d/nvm"  
  # This loads nvm bash_completion

# 5、然后使用 nvm 安装 node（完成后升级就👌了）

nvm install node

# vuepress 配置

# 1、创建并进入一个新目录

mkdir <#folder#> && cd <#folder#>

# 2、使用你喜欢的包管理器进行初始化

npm init -y

# 3、将 VuePress 安装为本地依赖,我们已经不再推荐全局安装 VuePress

npm install -D vuepress

# 4、创建你的第一篇文档

mkdir docs && echo '# Hello VuePress' > docs/README.md

# 5、在 package.json 中添加一些 scripts

这一步骤是可选的，但我们推荐你完成它。在下文中，我们会默认这些 scripts 已经被添加。

{
  "scripts": {
    "dev": "vuepress dev docs",
    "build": "vuepress build docs"
  }
}

# 6、在本地启动服务器

npm run dev

vuepress 会在 http://localhost:8080 启动一个热重载的开发服务器。

# 7、配置 config.js

.vuepress 文件夹下新建 config.js 文件

module.exports = {
  base: '/',
  title: '我的博客',
  description: 'Vuepress blog'
  themeConfig: {
        // 你的GitHub仓库，请正确填写
        repo: 'https://github.com/mrzhaohe/mrzhaohe.github.io',
        // 自定义仓库链接文字。
        repoLabel: 'My GitHub',
        nav: [
            { text: 'Home', link: '/' },
            { text: 'FirstBlog', link: '/blog/FirstBlog.md' },
 						{//带下拉的items
                text: 'other',
                items: [{
                    text: 'PHP',
                    link: '/other/PHP/'
                }, {
                    text: 'Flutter',
                    link: '/other/Flutter/'
                }]
            }
        ],
          sidebar: {
            '/other/PHP/': [{
                title: 'PHP',
                collapsable: false,
                children: [
                  '/'//可写可不写
                ]
            }],
            '/other/Flutter/': [{
                title: 'Flutter',
                collapsable: false,
                children: []
            }],
        },
        sidebarDepth: 1,//最深为1级标题
    }
}

# 部署到github

需要两个仓库，一个私有仓库用来存放源代码，另外一个以你名字开头的 name.github.io仓库

privite 仓库

name 仓库是编译之后的代码

push 到 mrzhaohe.github.io  执行 sh deploy.sh

本地查看  npm run  dev

# 1、创建脚本 deploy.sh

放置在根目录下

# 构建
vuepress build docs

# cd到构建输出目录
cd docs/.vuepress/dist

git init
git add -A
git commit -m '<#commit message#>'

# 发布到 https://<USERNAME>.github.io
git push -f https://github.com/mrzhaohe/mrzhaohe.github.io.git master

cd -

# 2、配置 package.json

{
  "name": "blog-of-zhaohe",
  "version": "1.0.0",
  "main": "index.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1",
    "dev": "vuepress dev docs",
    // 构建
    "build": "vuepress build docs",
    //发布
    "deploy": "npm run build && bash deploy.sh"
  },
  "keywords": [],
  "author": "",
  "license": "ISC",
  "description": "",
  "devDependencies": {
    "vuepress": "^1.5.4"
  }
}

# 3、创建一个私有仓库

存放你的源代码，当前文件夹推送到该私有仓库

# 4、创建以你名字开头的github.io仓库

# 5、部署

npm run deploy
